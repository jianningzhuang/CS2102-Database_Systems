CREATE OR REPLACE VIEW v10 (airport_name) AS

(SELECT n.continent, COUNT(DISTINCT a2.code) 
FROM Countries n, Airports a2, Connections c 
WHERE c.from_code = 'SIN' AND c.to_code = a2.code AND a2.country_iso2 = n.iso2 
GROUP BY n.continent;  
;
